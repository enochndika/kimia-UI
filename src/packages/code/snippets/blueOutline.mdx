function Code({ children }) {
  const [copied, setCopied] = React.useState(false);
  const resetCopy = React.useRef();

  const onCopyCode = () => {
    navigator.clipboard.writeText(children).then(() => setCopied(true));
  };

  React.useEffect(() => {
    if (copied) {
      resetCopy.current = setTimeout(() => setCopied(false), 2500);
    }

    return () => {
      clearTimeout(resetCopy.current);
    };
  }, [copied]);
  return (
    <pre className="inline-flex font-mono text-sm py-2.5 px-4 rounded overflow-x-auto max-w-full border border-blue-600 text-blue-600">
      <span>{children}</span>
      <span className="ml-5">
        {copied ? <CopiedIcon /> : <CopyIcon onClick={onCopyCode} />}
      </span>
    </pre>
  );
}
